let, const

function varTest() {
  var x = 1;
  if (true) {
    var x = 2;  // same variable!
  }
  console.log(x);  // 2
}

function letTest() {
  let x = 1;
  if (true) {
    let x = 2;  // different variable
  }
  console.log(x);  // 1
}

Qué es la programación funcional?

Un paradigma de programación

Un estilo de programación

Características

  • Eliminación de efectos secundarios
  • Funciones Puras
  • Inmutabilidad de los datos
  • Funciones de alto orden
  • Recursión

Efectos secundarios

Modifica alguna variable fuera de su scope

let a = 0;

function increment() {
	a++;
}

Funciones puras

  • Simpre da el mismo resultado para los mismos parametros
  • La llamada a la función no causa efectos secundarios
function addElement(someArray) {
	const anotherArray = someArray.slice();
	anotherArray.push("element");

	return anotherArray;
}

Funciones de alto orden

  • Tiene una o más funciones como parámetro
  • o retorna una función

Funciones de alto orden

function double(x) {
	return x*2;
}

function specialAdd(x, y, operation) {
	return operation(x) + operation(y);
}

specialAdd(2,3, double);
function Counter(){
	let count = 0;
	return function() {
		return ++count;
	}
}

let increment = Counter();

increment();
